建立 container:
docker run -it --mount source=opencv_ncnn,target=/home/app --name opencv_ncnn opencv_test

進入 docker:
docker exec -it opencv_ncnn bash

mkdir build && cd build
cmake -D CMAKE_BUILD_TYPE=RELEASE -D OPENCV_GENERATE_PKGCONFIG=ON -D CMAKE_INSTALL_PREFIX=/usr/local ..
make -j4
sudo make install
sudo ldconfig

測試:
cmake -D CMAKE_BUILD_TYPE=RELEASE -D OPENCV_GENERATE_PKGCONFIG=ON -D CMAKE_INSTALL_PREFIX=/usr/local ..

creat ncnn tool:
cd ncnn
mkdir -p build
cd build
cmake ..
make -j4
